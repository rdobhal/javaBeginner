class Solution {
    public int removeDuplicates(int[] nums) {
       // Input: nums = [1,1,2]
       // Output: 2, nums = [1,2,_] 
      if (nums.length == 0) return 0;
      int slow = 0;
      for(int fast = 1; fast < nums.length; fast++){
          if(nums[fast] != nums[slow]){
              slow++;
              nums[slow] = nums[fast]; 
          }
      }
          return slow + 1;
          }
}
